// Generate summary instantly
function generateSummaryReport(vendors) {
  const summary = {
    total: vendors.length,
    critical: vendors.filter(v => v.score >= 80).length,
    high: vendors.filter(v => v.score >= 60 && v.score < 80).length,
    medium: vendors.filter(v => v.score >= 40 && v.score < 60).length,
    low: vendors.filter(v => v.score < 40).length,
    avgScore: calculateAverage(vendors),
    topRisks: vendors.sort((a,b) => b.score - a.score).slice(0, 3)
  };
  
  displaySummaryReport(summary);
  
  // Show CTA for detailed report
  showDetailedReportCTA();
}
